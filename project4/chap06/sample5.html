if 조건문


    if (조건식) {
        조건식이 참일 때 수행할 코드
    }
    

    if (조건식) {
        조건식이 참일 때 수행할 코드
    } else {
        조건식이 거짓일 때 수행할 코드
    }
    

    if (조건식1) {
        조건식1 이 참일 때 수행할 코드
    } else if (조건식2) {
        조건식1 이 거짓이고 조건식2 가 참일 때 수행할 코드
    } else {
        조건식1, 조건식2 가 전부 거짓일 때 수행할 코드
    }

예제 코드


    if (x % 2 == 0) {
        document.getElementById("res1").innerHTML = "짝수 입니다.";
    } else {
        document.getElementById("res1").innerHTML = "홀수 입니다.";
    }

switch 문


    switch (식) {
        case 값1 :
            식의 결과가 값1에 매치되는 경우 수행할 코드
            break;
        case 값2 :
            식의 결과가 값2에 매치되는 경우 수행할 코드
            break;
        case 값3 :
            식의 결과가 값3에 매치되는 경우 수행할 코드
            break;
        default:
            식의 결과가 어떠한 case에도 매치되지 않는 경우 수행할 코드
    }

예제 코드


    switch (x) {
        case 1 :
            document.getElementById("res2").innerHTML = "1 입니다.";
            break;
        case 2 :
            document.getElementById("res2").innerHTML = "2 입니다.";
            break;
        default:
            document.getElementById("res2").innerHTML = "1 또는 2가 아닙니다.";
    }

for 문


    for(초기식; 조건식; 증감식) {
        반복 수행할 코드
    }

예제 코드


        
    for(var i = 0; i < x; i++) {
        document.getElementById("res3").innerHTML += i;
    }

while 문


    while(조건식) {
        조건식이 참일 때 반복 수행할 코드
    }

예제 코드


    var i = 0;
    while(i < x) {
        document.getElementById("res4").innerHTML += i;
        i++;
    }

for ... in 문


    for(변수명 in 객체) {
        객체가 가지는 속성에 대하여 반복을 진행
        반복 수행할 코드
    }

예제 코드


    var obj = {a: 1, b: 2, c: 3, d: 4}
    for(var name in obj) {
        document.getElementById("res5").innerHTML += name + ": " + obj[name] + "<br>"
    }

for ... of 문


    for(변수명 of 객체) {
        반복가능한 객체에 담긴 값을 사용하여 진행
        반복 수행할 코드
    }

예제 코드


    var obj = [1, 2, 3, 4]
    for(var v of obj) {
        document.getElementById("res6").innerHTML += v + "<br>"
    }

연산자

==, != 연산자

        값만 비교할 때 사용된다. 비교하는 두 값의 자료형과 관계없이
        값만 비교하는 경우 사용한다.

        1 == true -> true
        0 == false -> true
        undefined == null -> true

===, !== 연산자

        값과 자료형 모두 비교한다.

        1 === true -> false
        0 === false -> false
        undefined === null -> false
