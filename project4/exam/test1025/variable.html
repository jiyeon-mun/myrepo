function scope: var

function 단위


    function f1() {
        var x1 = 10;
        if(true) {
            var x2 = 20;
            console.log("if 안에서 x1 출력 -> " + x1); // 10
            console.log("if 안에서 x2 출력 -> " + x2); // 20
        }
        console.log("if 밖에서 x1 출력 -> " + x1); // 10
        console.log("if 밖에서 x2 출력 -> " + x2); // java: 접근불가 | javascript: 20
    }

block scope: let

{대괄호} 단위


    function f2() {
        var x1 = 10;
        if(true) {
            let x2 = 20;
            console.log("if 안에서 x1 출력 -> " + x1); // 10
            console.log("if 안에서 x2 출력 -> " + x2); // 20
        }
        console.log("if 밖에서 x1 출력 -> " + x1); // 10
        console.log("if 밖에서 x2 출력 -> " + x2); // javascript 접근불가
    }
