자바스크립트 객체

자바에서 사용하는 객체 처럼 객체에 속성과 메서드를 정의하여 사용할 수 있게 구현되어 있다. 단순하게는 Map 컬렉션 대용으로 사용할 수 있다.

예제 코드: 중괄호를 사용하여 객체 생성


    // 객체 생성
    var obj = {속성명: "속성값", 키: "값"};

    // 객체 속성에 접근하여 출력
    console.log(obj.속성명);
    console.log(obj["속성명"]); // 대괄호([]) 안에 큰따옴표("") 사용하기

    // 객체 속성에 접근하여 값 변경
    obj.속성명 = "변경할 값";
    obj["속성명"] = "변경할 값";

    // for ... in 반복문으로 접근하여 출력
    for(var k of obj) {
        console.log(k + ": " + oj=bj[k]);
    }

예제 코드: 함수와 new 생성자를 사용하여 객체 생성


    function func1(x, y) {
        var obj = {
            x: x, y: y
        }
        return obj;
    }

    function func2(x, y) {
        this.x = x;
        this.y = y;
    }

    var obj = new func1(10, 20);
    var obj = new func2(10, 20);

예제 코드: class 키워드로 객체 생성


    class func1 {
        constructor(x, y) {
            this.x = x;
            this.y = y;
        }
    }

    var obj = new func1(10, 20);

예제 코드: 객체에 메서드 생성


        function func1(x, y) {
            this.x = x;
            this.y = y;
            this.func1 = function() {
                console.log("메서드 실행 완료!");
            };
        }

        var obj = new func1(10, 20);
